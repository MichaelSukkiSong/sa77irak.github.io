class MyIter:
def init(self,li):
self.container=li
self.index=0
def iter(self):
return self
def next(self):

# 어느 시점에 StopIteration 에러만 넘겨줄 수 있으면 커스터마이징 조건 됨!!
if self.index >= len(self.container):
    raise StopIteration
ret=self.container[self.index]
self.index+=1
return ret

class Reader:
def init(self, filename):
self.f=open(filename, ‘rt’)
def iter(self):
return self
def next(self):
line =self.f.readline()
if line:
return line[:-1]
else:
self.f.close()
raise StopIteration

reader=Reader(‘test.txt’)
it=iter(reader)

coroutine function

: 함수 실행 도중에 실행 주도권을 다른 함수에 이양 가능하고
: “내가 원하는 시점에” 다시 실행 주도권을 가져올 수 있는 함수를 coroutine 함수라 함.
(non-preemptive하게 실행 가능)
->주도권을 주고 받을 수 있다는 뜻.
->yield를 만나는 순간 stackframe를 유지한채로 주도권을 다른 함수에 넘김. 양보한다? yield~
->파이썬 coroutine은 구현 기반이 generator
->그럴려면 stackframe이 여러개 있어야 될텐데, 이것은 heap에 저장되어 있기에 가능한 일~

def fibo_gen(n):
a, b = 0, 1
for _ in range(n):
yield a
a, b = b, a + b

fibo=fibo_gen(10)

next(fibo)
0
next(fibo)
1
next(fibo)
1
next(fibo)

for _ in range(10):
print(next(fibo), end=’ ‘)

def delegate():
g=gen()
print(“start”)
ret=yield from g # g부터 들어가서 다 조지고 나온다.
print(“end”)
print(f’return value : {ret}’)
return ret # 여기서는 stopiteration~ 에 value값 리턴

g=delegate()
